* China mirror
Add below lines in file: .doom.d/config.el

#+begin_src emacs_lisp
(setq package-archives
      '(("gnu" . "http://elpa.emacs-china.org/gnu/")
        ("melpa" . "http://elpa.emacs-china.org/melpa/")
        ("org" . "http://elpa.emacs-china.org/org/")))
#+end_src
